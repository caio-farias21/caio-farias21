class Dev {
    constructor(name, area, languages) {
        this.name = name;
        this.area = area;
        this.languages = languages;
    }
    getName() {
        return this.name;
    }
    getArea() {
        return this.area;
    }
    getLanguages() {
        return this.languages;
    }
}
